# agi3 bridge v3. Design: docs/architecture.md.
#
# Contract:
# - Python writes request files that ONLY set agi3_* numeric variables and
#   call agi3_consume_effect (request-only invariant, enforced Python-side).
# - agi3_consume_effect consumes a req_id at most once; redelivered requests
#   re-emit the result line for the last consumed id (idempotent ACK).
# - Console-level verbs (save/tick_day) cannot run from script. The runner
#   GUI fires FIXED command strings gated by agi3_want_* variables; this is
#   why wait has enumerated variants (7/30/90), not a free integer.
# - All emission goes through debug_log loc lines (agi3_l_english.yml).

agi3_init_effect = {
	set_variable = { name = agi3_last_consumed value = 0 }
	set_variable = { name = agi3_seq value = 0 }
	set_variable = { name = agi3_done value = 1 }
	set_variable = { name = agi3_evt_present value = 0 }
	set_variable = { name = agi3_want_save value = 0 }
	set_variable = { name = agi3_want_tick_7 value = 0 }
	set_variable = { name = agi3_want_tick_30 value = 0 }
	set_variable = { name = agi3_want_tick_90 value = 0 }
}

agi3_pulse = {
	if = {
		limit = { NOT = { exists = var:agi3_seq } }
		agi3_init_effect = yes
	}
	change_variable = { name = agi3_seq add = 1 }
	debug_log = "agi3_hb_line"
	debug_log_date = yes
}

agi3_emit_result_effect = {
	debug_log = "agi3_result_line"
}

agi3_consume_effect = {
	if = {
		limit = { NOT = { exists = var:agi3_last_consumed } }
		agi3_init_effect = yes
	}
	if = {
		# Redelivery of an already-consumed request: re-ACK, change nothing.
		# Result only if its staged work completed; ack while in flight.
		limit = {
			exists = var:agi3_req_id
			NOT = { var:agi3_req_id > var:agi3_last_consumed }
		}
		if = {
			limit = { var:agi3_done = 1 }
			agi3_emit_result_effect = yes
		}
		else = {
			debug_log = "agi3_ack_line"
		}
	}
	else_if = {
		limit = { exists = var:agi3_req_id }
		set_variable = { name = agi3_last_consumed value = var:agi3_req_id }
		agi3_dispatch_effect = yes
	}
}

agi3_dispatch_effect = {
	if = {
		# pulse.refresh (code 1): emit current facts.
		limit = { var:agi3_action = 1 }
		set_variable = { name = agi3_status value = 1 }
		set_variable = { name = agi3_done value = 1 }
		agi3_publish_state_effect = yes
		agi3_publish_gift_slots_effect = yes
		agi3_publish_camp_slots_effect = yes
		agi3_emit_result_effect = yes
		debug_log = "agi3_state_line"
	}
	else_if = {
		# checkpoint.save (code 2): stage; runner GUI fires the fixed
		# `save agi3_checkpoint` string; post-save run-file clears the flag.
		limit = { var:agi3_action = 2 }
		set_variable = { name = agi3_done value = 0 }
		set_variable = { name = agi3_want_save value = 1 }
		set_variable = { name = agi3_status value = 2 }
		debug_log = "agi3_ack_line"
	}
	else_if = {
		# wait.7/30/90 (code 3): one flag per enumerated variant, because the
		# runner GUI can only fire fixed command strings.
		limit = { var:agi3_action = 3 var:agi3_days = 7 }
		set_variable = { name = agi3_done value = 0 }
		set_variable = { name = agi3_want_tick_7 value = 1 }
		set_variable = { name = agi3_status value = 2 }
		debug_log = "agi3_ack_line"
	}
	else_if = {
		limit = { var:agi3_action = 3 var:agi3_days = 30 }
		set_variable = { name = agi3_done value = 0 }
		set_variable = { name = agi3_want_tick_30 value = 1 }
		set_variable = { name = agi3_status value = 2 }
		debug_log = "agi3_ack_line"
	}
	else_if = {
		limit = { var:agi3_action = 3 var:agi3_days = 90 }
		set_variable = { name = agi3_done value = 0 }
		set_variable = { name = agi3_want_tick_90 value = 1 }
		set_variable = { name = agi3_status value = 2 }
		debug_log = "agi3_ack_line"
	}
	else_if = {
		# decision.probe/take (code 7): allowlist index in agi3_slot.
		limit = { var:agi3_action = 7 }
		set_variable = { name = agi3_done value = 1 }
		if = {
			limit = { var:agi3_verb = 1 }
			agi3_decision_probe_effect = yes
		}
		else = {
			agi3_decision_execute_effect = yes
		}
	}
	else_if = {
		# move_camp.probe/move (code 4): fixed-title-holder slots.
		limit = { var:agi3_action = 4 }
		set_variable = { name = agi3_done value = 1 }
		if = {
			limit = { var:agi3_verb = 1 }
			agi3_camp_probe_effect = yes
		}
		else = { agi3_camp_execute_effect = yes }
	}
	else_if = {
		# gift.probe/send (code 5): courtier slots.
		limit = { var:agi3_action = 5 }
		set_variable = { name = agi3_done value = 1 }
		if = {
			limit = { var:agi3_verb = 1 }
			agi3_gift_probe_effect = yes
		}
		else = { agi3_gift_execute_effect = yes }
	}
	else_if = {
		# event_option.select (code 13): arm the deployed v2 generated-GUI
		# selector (proven live in v2). The GUI gates on EventWindow.IsOnTop,
		# EventOption.IsValid, NOT dangerous-flagged, index match; the
		# scripted_gui consume-recorder closes the loop with our result line.
		limit = { var:agi3_action = 13 }
		set_variable = { name = agi3_done value = 0 }
		set_variable = { name = agi_ck3_bridge_v2_request_id value = var:agi3_req_id }
		set_variable = { name = agi_ck3_bridge_v2_action_type value = 16 }
		set_variable = { name = agi_ck3_bridge_event_option_request_present value = 1 }
		set_variable = { name = agi_ck3_bridge_event_option_requested_event_id value = var:agi3_evt_event_id }
		set_variable = { name = agi_ck3_bridge_event_option_requested_index value = var:agi3_slot }
		set_variable = { name = agi_ck3_bridge_event_option_requested_option_code value = 1 }
		debug_log = "agi3_ack_line"
	}
	else = {
		# Unported family: reject explicitly. Never silently absorb.
		set_variable = { name = agi3_status value = 9 }
		set_variable = { name = agi3_done value = 1 }
		debug_log = "agi3_reject_line"
	}
}

# Fired by the static post-save run file (run agi3_after_save.txt) that the
# runner chains after `save agi3_checkpoint`.

	set_variable = { name = agi3_want_save value = 0 }
	set_variable = { name = agi3_done value = 1 }
	agi3_emit_result_effect = yes
}

	debug_log_date = yes
}

# Slot publication: MECHANICAL criteria only — courtiers for
# gift, fixed title-holders for move_camp. Never strategic curation: the
# harness must not hint. Published on pulse.refresh dispatch, not on the
# at-rest heartbeat (log economy). slot_clear flushes stale slots AND
# revokes probe grants Python-side.

agi3_publish_gift_slots_effect = {
	debug_log = "agi3_slot_clear_gift_line"
	set_variable = { name = agi3_pub_count value = 0 }
	every_courtier = {
		limit = { NOT = { this = root } root.var:agi3_pub_count < 16 }

			limit = { root.var:agi3_pub_count = 0 }
			root = { set_variable = { name = agi3_gift_slot_0 value = prev } }

# Claim protocol: any visible layer may fire staged work; the FIRST chain
# claims (flag 1->2), hiding every fire-widget copy. Frozen layers cannot
# double-fire because hidden/inactive widgets do not animate - the same
# property that stalls them. After-widgets (flag==2) complete from any
# live layer; completion effects are idempotent.
agi3_claim_save_effect = {
	if = { limit = { var:agi3_want_save = 1 } set_variable = { name = agi3_want_save value = 2 } }
}
agi3_claim_t7_effect = {
	if = { limit = { var:agi3_want_tick_7 = 1 } set_variable = { name = agi3_want_tick_7 value = 2 } }
}
agi3_claim_t30_effect = {
	if = { limit = { var:agi3_want_tick_30 = 1 } set_variable = { name = agi3_want_tick_30 value = 2 } }
}
agi3_claim_t90_effect = {
	if = { limit = { var:agi3_want_tick_90 = 1 } set_variable = { name = agi3_want_tick_90 value = 2 } }
}

# Ladder primitives, computed in-engine where scopes are native truth.
# Mechanical facts only — no strategy hints (fairness commitment).
agi3_publish_state_effect = {
	set_variable = { name = agi3_f_gold value = gold }
	set_variable = { name = agi3_f_prestige value = prestige }
	set_variable = { name = agi3_f_tier value = highest_held_title_tier }
	set_variable = { name = agi3_f_hre value = 0 }
	if = {
		limit = { exists = title:e_hre.holder title:e_hre.holder = root }
		set_variable = { name = agi3_f_hre value = 1 }
	}
	# Landless adventurers hold a duchy-tier TITULAR title; tier alone
	# falsely reads as landed (caught live: 25 ghost points for a tent).
	set_variable = { name = agi3_f_landless value = 0 }
	if = {
		limit = { has_government = landless_adventurer_government }
		set_variable = { name = agi3_f_landless value = 1 }
	}
}

# Event detection via layer-liveness: event-window hosts run agi3_event_open,
# the hud host runs agi3_event_clear — whichever layer animates tells the
# truth. Transition-guarded to avoid log spam at 2 Hz.
agi3_event_open_effect = {
	if = {
		limit = { NOT = { var:agi3_evt_present = 1 } }
		set_variable = { name = agi3_evt_present value = 1 }
		debug_log = "agi3_event_line"
	}
}
agi3_event_clear_effect = {
	if = {
		limit = { var:agi3_evt_present = 1 }
		set_variable = { name = agi3_evt_present value = 0 }
		debug_log = "agi3_event_line"
	}
}
